 32 /**
 33 * Constructs a new Container instance.
 34 * @class Container are nestable display lists that allow you to work with compound display elements. For example you could group arm, leg, torso and head Bitmaps together into a Person Container, and transform them as a group, while still being able to move the individual parts relative to each other. Children of containers have their transform and alpha properties concatenated with their parent Container. For example, a Shape with x=100 and alpha=0.5, placed in a Container with x=50 and alpha=0.7 will be rendered to the canvas at x=150 and alpha=0.35. Containers have some overhead, so you generally shouldn't create a Container to hold a single child.
 35 * @augments DisplayObject
 36 **/
 37 Container = function() {
 38   this.initialize();
 39 }
 40 Container.prototype = new DisplayObject();
 41 
 42 // public properties:
 43 	/** The array of children in the display list. You should usually use the child management methods, rather than accessing this directly, but it is included for advanced users. */
 44 	Container.prototype.children = null;

 57 	Container.prototype.isVisible = function() {
 58 		return this.visible && this.alpha > 0 && this.children.length && this.scaleX != 0 && this.scaleY != 0;
 59 	}

 61 	/** @private */
 62 	Container.prototype.DisplayObject_draw = Container.prototype.draw;
 63 	Container.prototype.draw = function(ctx, ignoreCache, _mtx) {
 64 		var snap = Stage._snapToPixelEnabled;
 65 		if (!_mtx) {
 66 			_mtx = new Matrix2D();
 67 			_mtx.appendProperties(this.alpha, this.shadow, this.compositeOperation);
 68 		}
 69 		if (this.DisplayObject_draw(ctx, ignoreCache)) { return true; }
 70 		var l = this.children.length;
 71 		// this ensures we don't have issues with display list changes that occur during a draw:
 72 		var list = this.children.slice(0);
 73 		for (var i=0; i<l; i++) {
 74 			var child = list[i];
 75 			if (child.tick) { child.tick(); }
 76 			if (!child.isVisible()) { continue; }
 77 
 78 			var mtx = _mtx.clone();
 79 			mtx.appendTransform(child.x, child.y, child.scaleX, child.scaleY, child.rotation, child.skewX, child.skewY, child.regX, child.regY);
 80 			mtx.appendProperties(child.alpha, child.shadow, child.compositeOperation);
 81 
 82 			if (!(child instanceof Container)) {
 83 				if (snap && child.snapToPixel && mtx.a == 1 && mtx.b == 0 && mtx.c == 0 && mtx.d == 1) {
 84 					ctx.setTransform(mtx.a,  mtx.b, mtx.c, mtx.d, mtx.tx+0.5|0, mtx.ty+0.5|0);
 85 				} else {
 86 					ctx.setTransform(mtx.a,  mtx.b, mtx.c, mtx.d, mtx.tx, mtx.ty);
 87 				}
 88 				ctx.globalAlpha = mtx.alpha;
 89 				ctx.globalCompositeOperation = mtx.compositeOperation || "source-over";
 90 				if (mtx.shadow) { this.applyShadow(ctx, mtx.shadow); }
 91 			}
 92 			child.draw(ctx, false, mtx);
 93 		}
 94 		return true;
 95 	}

 97 	/**
 98 	* Adds a child to the top of the display list. You can also add multiple children, such as "addChild(child1, child2, ...);". Returns the child that was added, or the last child if multiple children were added.
 99 	* @param child The display object to add.
100 	**/
101 	Container.prototype.addChild = function(child) {
102 		var l = arguments.length;
103 		if (l > 1) {
104 			for (var i=0; i<l; i++) { this.addChild(arguments[i]); }
105 			return arguments[l-1];
106 		}
107 		if (child.parent) { child.parent.removeChild(child); }
108 		child.parent = this;
109 		this.children.push(child);
110 		return child;
111 	}

113 	/**
114 	* Adds a child to the display list at the specified index, bumping children at equal or greater indexes up one, and setting its parent to this Container. You can also add multiple children, such as "addChildAt(child1, child2, ..., index);". The index must be between 0 and numChildren. For example, to add myShape under otherShape in the display list, you could use: container.addChildAt(myShape, container.getChildIndex(otherShape)). This would also bump otherShape's index up by one. Returns the last child that was added, or the last child if multiple children were added.
115 	* @param child The display object to add.
116 	* @param index The index to add the child at.
117 	**/
118 	Container.prototype.addChildAt = function(child, index) {
119 		var l = arguments.length;
120 		if (l > 2) {
121 			index = arguments[i-1];
122 			for (var i=0; i<l-1; i++) { this.addChildAt(arguments[i], index+i); }
123 			return arguments[l-2];
124 		}
125 		if (child.parent) { child.parent.removeChild(child); }
126 		child.parent = this;
127 		this.children.splice(index, 0, child);
128 		return child;
129 	}

131 	/**
132 	* Removes the specified child from the display list. Note that it is faster to use removeChildAt() if the index is already known. You can also remove multiple children, such as "removeChild(child1, child2, ...);". Returns true if the child (or children) was removed, or false if it was not in the display list.
133 	* @param child The child to remove.
134 	**/
135 	Container.prototype.removeChild = function(child) {
136 		var l = arguments.length;
137 		if (l > 1) {
138 			var good = true;
139 			for (var i=0; i<l; i++) { good = good && this.removeChild(arguments[i]); }
140 			return good;
141 		}
142 		return this.removeChildAt(this.children.indexOf(child));
143 	}
144 	
145 	/**
146 	* Removes the child at the specified index from the display list, and sets its parent to null. You can also remove multiple children, such as "removeChildAt(2, 7, ...);". Returns true if the child (or children) was removed, or false if any index was out of range.
147 	* @param The index of the child to remove.
148 	**/
149 	Container.prototype.removeChildAt = function(index) {
150 		var l = arguments.length;
151 		if (l > 1) {
152 			var a = [];
153 			for (var i=0; i<l; i++) { a[i] = arguments[i]; }
154 			a.sort(function(a, b) { return b-a; })
155 			var good = true;
156 			for (var i=0; i<l; i++) { good = good && this.removeChildAt(a[i]); }
157 			return good;
158 		}
159 		if (index < 0 || index > this.children.length-1) { return false; }
160 		var child = this.children[index];
161 		if (child != null) { child.parent = null; }
162 		this.children.splice(index, 1);
163 		return true;
164 	}

167 	* Removes all children from the display list.
168 	**/
169 	Container.prototype.removeAllChildren = function() {
170 		while (this.children.length) { this.removeChildAt(0); }
171 	}

205 	* Returns true if the specified display object either is this container or is a descendent
206 	* (child, grandchild, etc) of this container.

208 	Container.prototype.contains = function(child) {
209 		while (child) {
210 			if (child == this) { return true; }
211 			child = child.parent;
212 		}
213 		return false;
214 	}

217 		// TODO: optimize to use the fast cache check where possible.
218 		return (this.getObjectUnderPoint(x, y) != null);

221 	/**
222 	* Returns an array of all display objects under the specified coordinates that are in this container's display list. This routine ignores any display objects with mouseEnabled set to false. The array will be sorted in order of visual depth, with the top-most display object at index 0. This uses shape based hit detection, and can be an expensive operation to run, so it is best to use it carefully. For example, if testing for objects under the mouse, test on tick (instead of on mousemove), and only if the mouse's position has changed.
223 	* @param x The x position in the container to test.
224 	* @param y The y position in the container to test.
225 	**/
226 	Container.prototype.getObjectsUnderPoint = function(x, y) {
227 		var arr = [];
228 		var pt = this.localToGlobal(x, y);
229 		this._getObjectsUnderPoint(pt.x, pt.y, arr);
230 		return arr;
231 	}
232 
233 	/**
234 	* Similar to getObjectsUnderPoint(), but returns only the top-most display object. This runs significantly faster than getObjectsUnderPoint(), but is still an expensive operation. See getObjectsUnderPoint() for more information.
235 	* @param x The x position in the container to test.
236 	* @param y The y position in the container to test.
237 	**/
238 	Container.prototype.getObjectUnderPoint = function(x, y) {
239 		var pt = this.localToGlobal(x, y);
240 		return this._getObjectsUnderPoint(pt.x, pt.y);
241 	}

243 	/**
244 	* Returns a clone of this Container. Some properties that are specific to this instance's current context are reverted to their defaults (for example .parent).
245 	* @param recursive If true, all of the descendants of this container will be cloned recursively. If false, the properties of the container will be cloned, but the new instance will not have any children.
246 	**/
247 	Container.prototype.clone = function(recursive) {
248 		var o = new Container();
249 		this.cloneProps(o);
250 		if (recursive) {
251 			var arr = o.children = [];
252 			for (var i=0, l=this.children.length; i<l; i++) {
253 				arr.push(this.children[i].clone(recursive));
254 			}
255 		}
256 		return o;
257 	}

263 // private properties:
264 	/** @private */
265 	Container.prototype._getObjectsUnderPoint = function(x, y, arr, mouseEvents) {
266 
267 		var ctx = DisplayObject._hitTestContext;
268 		var canvas = DisplayObject._hitTestCanvas;
269 		var mtx = DisplayObject._workingMatrix;
270 		var hasHandler = mouseEvents && (this.onPress || this.onClick);
271 
272 		// if we have a cache handy, we can use it to do a quick check:
273 		if (this.cacheCanvas) {
274 			this.getConcatenatedMatrix(mtx);
275 			ctx.setTransform(mtx.a,  mtx.b, mtx.c, mtx.d, mtx.tx-x, mtx.ty-y);
276 			ctx.globalAlpha = mtx.alpha;
277 			this.draw(ctx);
278 			if (this._testHit(ctx)) {
279 				canvas.width = 0;
280 				canvas.width = 1;
281 				if (hasHandler) { return this; }
282 			} else {
283 				return null;
284 			}
285 		}
286 		
287 		// draw children one at a time, and check if we get a hit:
288 		var l = this.children.length;
289 		for (var i=l-1; i>=0; i--) {
290 			var child = this.children[i];
291 			if (!child.isVisible() || !child.mouseEnabled) { continue; }
292 			
293 			if (child instanceof Container) {
294 				var result;
295 				if (hasHandler) {
296 					// only concerned about the first hit, because this container is going to claim it anyway:
297 					result = child._getObjectsUnderPoint(x, y);
298 					if (result) { return this; }
299 				} else {
300 					result = child._getObjectsUnderPoint(x, y, arr, mouseEvents);
301 					if (!arr && result) { return result; }
302 				}
303 			} else if (!mouseEvents || (mouseEvents && (child.onPress || child.onClick || hasHandler))) {
304 				child.getConcatenatedMatrix(mtx);
305 				ctx.setTransform(mtx.a,  mtx.b, mtx.c, mtx.d, mtx.tx-x, mtx.ty-y);
306 				ctx.globalAlpha = mtx.alpha;
307 				child.draw(ctx);
308 				if (!this._testHit(ctx)) { continue; }
309 				canvas.width = 0;
310 				canvas.width = 1;
311 				if (hasHandler) { return this; }
312 				else if (arr) { arr.push(child); }
313 				else { return child; }
314 			}
315 		}
316 		return null;
317 	}
