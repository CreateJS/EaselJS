 30 (function(window) {
 31 
 32 /**
 33 * Constructs a new Text instance.
 34 * @param text Optional. The text to display.
 35 * @param font Optional. The font style to use. Any valid value for the CSS font attribute is acceptable (ex. "36px bold Arial").
 36 * @param color Optional. The color to draw the text in. Any valid value for the CSS color attribute is acceptable (ex. "#F00").
 37 * @class Allows you to display one or more lines of dynamic text (not user editable) in the display list. Line wrapping
 38  * support (using the lineWidth is very basic, wrapping on spaces and tabs only. Note that as an alternative to Text,
 39  * you can position HTML text above or below the canvas relative to items in the display list using the localToGlobal() method.
 40 * @augments DisplayObject
 41 **/
 42 Text = function(text, font, color) {
 43   this.initialize(text, font, color);
 44 }
 45 Text.prototype = new DisplayObject();
 46 
 47 /** @private */
 48 var canvas = document.createElement("canvas");
 49 Text._workingContext = canvas.getContext("2d");
 50 
 51 // public properties:
 52 	/** The text to display. */
 53 	Text.prototype.text = "";
 54 	/** The font style to use. Any valid value for the CSS font attribute is acceptable (ex. "bold 36px Arial"). */
 55 	Text.prototype.font = null;
 56 	/** The color to draw the text in. Any valid value for the CSS color attribute is acceptable (ex. "#F00"). */
 57 	Text.prototype.color = null;
 58 	/** The horizontal text alignment. Any of start, end, left, right, and center. For detailed information view the <a href="http://www.whatwg.org/specs/web-apps/current-work/multipage/the-canvas-element.html#text-0">whatwg spec</a>. */
 59 	Text.prototype.textAlign = null;
 60 	/** The vertical alignment point on the font. Any of top, hanging, middle, alphabetic, ideographic, or bottom. For detailed information view the <a href="http://www.whatwg.org/specs/web-apps/current-work/multipage/the-canvas-element.html#text-0">whatwg spec</a>. */
 61 	Text.prototype.textBaseline = null;
 62 	/** The maximum width to draw the text. If maxWidth is specified (not null), the text will be condensed or shrunk to make it fit in this width. For detailed information view the <a href="http://www.whatwg.org/specs/web-apps/current-work/multipage/the-canvas-element.html#text-0">whatwg spec</a>. */
 63 	Text.prototype.maxWidth = null;
 64 	/** If true, the text will be drawn as a stroke (outline). If false, the text will be drawn as a fill. */
 65 	Text.prototype.outline = false;
 66 	/** Indicates the line height (vertical distance between baselines) for multi-line text. If null, the value of getMeasuredLineHeight is used. */
 67 	Text.prototype.lineHeight = null;
 68 	/** Indicates the maximum width for a line of text before it is wrapped to multiple lines. If null, the text will not be wrapped. */
 69 	Text.prototype.lineWidth = null;
 70 	
 71 // constructor:
 72 	/** @ignore */
 73 	Text.prototype.DisplayObject_initialize = Text.prototype.initialize;
 74 	/** @ignore */
 75 	Text.prototype.initialize = function(text, font, color) {
 76 		this.DisplayObject_initialize();
 77 		this.text = text;
 78 		this.font = font;
 79 		this.color = color ? color : "#000";
 80 	}
 81 	
 82 // public methods:
 83 	Text.prototype.isVisible = function() {
 84 		return Boolean(this.visible && this.alpha > 0 && this.scaleX != 0 && this.scaleY != 0 && this.text != null && this.text != "");
 85 	}
 86 
 87 	Text.prototype.DisplayObject_draw = Text.prototype.draw;
 88 	Text.prototype.draw = function(ctx, ignoreCache) {
 89 		if (this.DisplayObject_draw(ctx, ignoreCache)) { return true; }
 90 		
 91 		if (this.outline) { ctx.strokeStyle = this.color; }
 92 		else { ctx.fillStyle = this.color; }
 93 		ctx.font = this.font;
 94 		ctx.textAlign = this.textAlign ? this.textAlign : "start";
 95 		ctx.textBaseline = this.textBaseline ? this.textBaseline : "alphabetic";
 96 
 97 		var lines = String(this.text).split(/(?:\r\n|\r|\n)/);
 98 		var lineHeight = (this.lineHeight == null) ? this.getMeasuredLineHeight() : this.lineHeight;
 99 		var y = 0;
100 		for (var i=0, l=lines.length; i<l; i++) {
101 			var w = ctx.measureText(lines[i]).width;
102 			if (this.lineWidth == null || w < this.lineWidth) {
103 				this._drawTextLine(ctx, lines[i], y);
104 				y += lineHeight;
105 				continue;
106 			}
107 
108 			// split up the line
109 			var words = lines[i].split(/(\s)/);
110 			var str = words[0];
111 			for (var j=1, jl=words.length; j<jl; j+=2) {
112 				// Line needs to wrap:
113 				if (ctx.measureText(str + words[j] + words[j+1]).width > this.lineWidth) {
114 					this._drawTextLine(ctx, str, y);
115 					y += lineHeight;
116 					str = words[j+1];
117 				} else {
118 					str += words[j] + words[j+1];
119 				}
120 			}
121 			this._drawTextLine(ctx, str, y); // Draw remaining text
122 			y += lineHeight;
123 		}
124 		return true;
125 	}
126 	
127 	/**
128 	* Returns the measured, untransformed width of the text.
129 	**/
130 	Text.prototype.getMeasuredWidth = function() {
131 		return this._getWorkingContext().measureText(this.text).width;
132 	}
133 
134 	/**
135 	 * Returns an approximate line height of the text, ignoring the lineHeight property. This is based on the measured width of
136 	 * a "M" character multiplied by 1.2, which approximates em for most fonts.
137 	 */
138 	Text.prototype.getMeasuredLineHeight = function() {
139 		return this._getWorkingContext().measureText("M").width*1.2;
140 	}
141 	
142 	Text.prototype.clone = function() {
143 		var o = new Text(this.text, this.font, this.color);
144 		this.cloneProps(o);
145 		return o;
146 	}
147 		
148 	Text.prototype.toString = function() {
149 		return "[Text (text="+  (this.text.length > 20 ? this.text.substr(0, 17)+"..." : this.text) +")]";
150 	}
151 	
152 // private methods:
153 	
154 	/** @private */
155 	Text.prototype.DisplayObject_cloneProps = Text.prototype.cloneProps;
156 	/** @private */
157 	Text.prototype.cloneProps = function(o) {
158 		this.DisplayObject_cloneProps(o);
159 		o.textAlign = this.textAlign;
160 		o.textBaseline = this.textBaseline;
161 		o.maxWidth = this.maxWidth;
162 		o.outline = this.outline;
163 		o.lineHeight = this.lineHeight;
164 		o.lineWidth = this.lineWidth;
165 	}
166 
167 	Text.prototype._getWorkingContext = function() {
168 		var ctx = Text._workingContext;
169 		ctx.font = this.font;
170 		ctx.textAlign = this.textAlign ? this.textAlign : "start";
171 		ctx.textBaseline = this.textBaseline ? this.textBaseline : "alphabetic";
172 		return ctx;
173 	}
174 	
175 	Text.prototype._drawTextLine = function(ctx, text, y) {
176 		if (this.outline) { ctx.strokeText(text, 0, y, this.maxWidth); }
177 		else { ctx.fillText(text, 0, y, this.maxWidth); }
178 	}
179 
180 window.Text = Text;
181 }(window));
