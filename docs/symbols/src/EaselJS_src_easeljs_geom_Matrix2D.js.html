 32 /**
 33 * Constructs a new Matrix2D instance.
 34 * @param a Specifies the a property for the new matrix.
 35 * @param b Specifies the b property for the new matrix.
 36 * @param c Specifies the c property for the new matrix.
 37 * @param d Specifies the d property for the new matrix.
 38 * @param tx Specifies the tx property for the new matrix.
 39 * @param ty Specifies the ty property for the new matrix.
 40 * @class Represents an affine tranformation matrix, and provides tools for constructing and concatenating matrixes.
 41 **/
 42 Matrix2D = function(a, b, c, d, tx, ty) {
 43   this.initialize(a, b, c, d, tx, ty);

 77 // constructor:
 78 	/** @ignore */
 79 	Matrix2D.prototype.initialize = function(a, b, c, d, tx, ty) {
 80 		if (a != null) { this.a = a; }
 81 		if (b != null) { this.b = b; }
 82 		if (c != null) { this.c = c; }
 83 		if (d != null) { this.d = d; }
 84 		if (tx != null) { this.tx = tx; }
 85 		if (ty != null) { this.ty = ty; }
 86 	}

 88 // public methods:
 89 	/**
 90 	* Concatenates the specified matrix properties with this matrix. You must provide values for all of the parameters.
 91 	**/
 92 	Matrix2D.prototype.prepend = function(a, b, c, d, tx, ty) {
 93 		var tx1 = this.tx;
 94 		if (a != 1 || b != 0 || c != 0 || d != 1) {
 95 			var a1 = this.a;
 96 			var c1 = this.c;
 97 			this.a  = a1*a+this.b*c;
 98 			this.b  = a1*b+this.b*d;
 99 			this.c  = c1*a+this.d*c;
100 			this.d  = c1*b+this.d*d;
101 		}
102 		this.tx = tx1*a+this.ty*c+tx;
103 		this.ty = tx1*b+this.ty*d+ty;
104 	}
105 
106 	Matrix2D.prototype.append = function(a, b, c, d, tx, ty) {
107 		var a1 = this.a;
108 		var b1 = this.b;
109 		var c1 = this.c;
110 		var d1 = this.d;
111 		
112 		this.a  = a*a1+b*c1;
113 		this.b  = a*b1+b*d1;
114 		this.c  = c*a1+d*c1;
115 		this.d  = c*b1+d*d1;
116 		this.tx = tx*a1+ty*c1+this.tx;
117 		this.ty = tx*b1+ty*d1+this.ty;
118 	}

121 	* Prepends the specified matrix with this matrix.
122 	**/
123 	Matrix2D.prototype.prependMatrix = function(matrix) {
124 		this.prepend(matrix.a, matrix.b, matrix.c, matrix.d, matrix.tx, matrix.ty);
125 		this.prependProperties(matrix.alpha, matrix.shadow,  matrix.compositeOperation);
126 	}

129 	* Appends the specified matrix with this matrix.
130 	**/
131 	Matrix2D.prototype.appendMatrix = function(matrix) {
132 		this.append(matrix.a, matrix.b, matrix.c, matrix.d, matrix.tx, matrix.ty);
133 		this.appendProperties(matrix.alpha, matrix.shadow,  matrix.compositeOperation);
134 	}

136 	/**
137 	 * Generates matrix properties from the specified display object transform properties, and prepends them with this matrix.
138 	 * For example, you can use this to generate a matrix from a display object: var mtx = new Matrix2D(); mtx.prependTransform(o.x, o.y, o.scaleX, o.scaleY, o.rotation);
139 	 * @param x
140 	 * @param y
141 	 * @param scaleX
142 	 * @param scaleY
143 	 * @param rotation
144 	 * @param skewX
145 	 * @param skewY
146 	 * @param regX Optional.
147 	 * @param regY Optional.
148 	**/
149 	Matrix2D.prototype.prependTransform = function(x, y, scaleX, scaleY, rotation, skewX, skewY, regX, regY) {
150 		if (rotation%360) {
151 			var r = rotation*Matrix2D.DEG_TO_RAD;
152 			var cos = Math.cos(r);
153 			var sin = Math.sin(r);
154 		} else {
155 			cos = 1;
156 			sin = 0;
157 		}
158 		
159 		if (regX || regY) {
160 			// append the registration offset:
161 			this.tx -= regX; this.ty -= regY;
162 		}
163 		if (skewX || skewY) {
164 			// TODO: can this be combined into a single prepend operation?
165 			skewX *= Matrix2D.DEG_TO_RAD;
166 			skewY *= Matrix2D.DEG_TO_RAD;
167 			this.prepend(cos*scaleX, sin*scaleX, -sin*scaleY, cos*scaleY, 0, 0);
168 			this.prepend(Math.cos(skewY), Math.sin(skewY), -Math.sin(skewX), Math.cos(skewX), x, y);
169 		} else {
170 			this.prepend(cos*scaleX, sin*scaleX, -sin*scaleY, cos*scaleY, x, y);
171 		}
172 
173 	}
174 
175 	/**
176 	 * Generates matrix properties from the specified display object transform properties, and appends them with this matrix.
177 	 * For example, you can use this to generate a matrix from a display object: var mtx = new Matrix2D(); mtx.appendTransform(o.x, o.y, o.scaleX, o.scaleY, o.rotation);
178 	 * @param x
179 	 * @param y
180 	 * @param scaleX
181 	 * @param scaleY
182 	 * @param rotation
183 	 * @param skewX
184 	 * @param skewY
185 	 * @param regX Optional.
186 	 * @param regY Optional.
187 	**/
188 	Matrix2D.prototype.appendTransform = function(x, y, scaleX, scaleY, rotation, skewX, skewY, regX, regY) {
189 		if (rotation%360) {
190 			var r = rotation*Matrix2D.DEG_TO_RAD;
191 			var cos = Math.cos(r);
192 			var sin = Math.sin(r);
193 		} else {
194 			cos = 1;
195 			sin = 0;
196 		}
197 
198 		if (skewX || skewY) {
199 			// TODO: can this be combined into a single append?
200 			skewX *= Matrix2D.DEG_TO_RAD;
201 			skewY *= Matrix2D.DEG_TO_RAD;
202 			this.append(Math.cos(skewY), Math.sin(skewY), -Math.sin(skewX), Math.cos(skewX), x, y);
203 			this.append(cos*scaleX, sin*scaleX, -sin*scaleY, cos*scaleY, 0, 0);
204 		} else {
205 			this.append(cos*scaleX, sin*scaleX, -sin*scaleY, cos*scaleY, x, y);
206 		}
207 
208 		if (regX || regY) {
209 			// prepend the registration offset:
210 			this.tx -= regX*this.a+regY*this.c;
211 			this.ty -= regX*this.b+regY*this.d;
212 		}
213 	}

215 	/**
216 	 * Applies a rotation transformation to the matrix.
217 	 * @param angle The angle in degrees.
218 	 **/
219 	Matrix2D.prototype.rotate = function(angle) {
220 		var cos = Math.cos(angle);
221 		var sin = Math.sin(angle);
222 		
223 		var a1 = this.a;
224 		var c1 = this.c;
225 		var tx1 = this.tx;
226 		
227 		this.a = a1*cos-this.b*sin;
228 		this.b = a1*sin+this.b*cos;
229 		this.c = c1*cos-this.d*sin;
230 		this.d = c1*sin+this.d*cos;
231 		this.tx = tx1*cos-this.ty*sin;
232 		this.ty = tx1*sin+this.ty*cos;
233 	}

235 	/**
236 	 * Applies a skew transformation to the matrix.
237 	 * @param skewX The amount to skew horizontally in degrees.
238 	 * @param skewY The amount to skew vertically in degrees.
239 	 */
240 	Matrix2D.prototype.skew = function(skewX, skewY) {
241 		skewX = skewX*Matrix2D.DEG_TO_RAD;
242 		skewY = skewY*Matrix2D.DEG_TO_RAD;
243 		this.append(Math.cos(skewY), Math.sin(skewY), -Math.sin(skewX), Math.cos(skewX), 0, 0);
244 	}

247 	* Applies a scale transformation to the matrix.
248 	**/
249 	Matrix2D.prototype.scale = function(x, y) {
250 		this.a *= x;
251 		this.d *= y;
252 		this.tx *= x;
253 		this.ty *= y;
254 	}

256 	/**
257 	* Translates the matrix on the x and y axes.
258 	**/
259 	Matrix2D.prototype.translate = function(x, y) {
260 		this.tx += x;
261 		this.ty += y;
262 	}

264 	/**
265 	* Sets the properties of the matrix to those of an identity matrix (one that applies a null transformation).
266 	**/
267 	Matrix2D.prototype.identity = function() {
268 		this.alpha = this.a = this.d = 1;
269 		this.b = this.c = this.tx = this.ty = 0;
270 		this.shadow = this.compositeOperation = null;
271 	}

273 	/**
274 	* Inverts the matrix, causing it to perform the opposite transformation.
275 	**/
276 	Matrix2D.prototype.invert = function() {
277 		var a1 = this.a;
278 		var b1 = this.b;
279 		var c1 = this.c;
280 		var d1 = this.d;
281 		var tx1 = this.tx;
282 		var n = a1*d1-b1*c1;
283 		
284 		this.a = d1/n;
285 		this.b = -b1/n;
286 		this.c = -c1/n;
287 		this.d = a1/n;
288 		this.tx = (c1*this.ty-d1*tx1)/n;
289 		this.ty = -(a1*this.ty-b1*tx1)/n;
290 	}

292 	/**
293 	 * Decomposes the matrix into transform properties (x, y, scaleX, scaleY, and rotation). Note that this these values
294 	 * may not match the transform properties you used to generate the matrix, though they will produce the same visual
295 	 * results.
296 	 * @param target The object to apply the transform properties to. If null, then a new object will be returned.
297 	 */
298 	Matrix2D.prototype.decompose = function(target) {
299 		// TODO: it would be nice to be able to solve for whether the matrix can be decomposed into only scale/rotation even when scale is negative
300 		if (target == null) { target = {}; }
301 		target.x = this.tx;
302 		target.y = this.ty;
303 		target.scaleX = Math.sqrt(this.a * this.a + this.b * this.b);
304 		target.scaleY = Math.sqrt(this.c * this.c + this.d * this.d);
305 
306 		var skewX = Math.atan2(-this.c, this.d);
307 		var skewY = Math.atan2(this.b, this.a);
308 
309 		if (skewX == skewY) {
310 			target.rotation = skewY/Matrix2D.DEG_TO_RAD;
311 			if (this.a < 0 && this.d >= 0) {
312 				target.rotation += (target.rotation <= 0) ? 180 : -180;
313 			}
314 			target.skewX = target.skewY = 0;
315 		} else {
316 			target.skewX = skewX/Matrix2D.DEG_TO_RAD;
317 			target.skewY = skewY/Matrix2D.DEG_TO_RAD;
318 		}
319 		return target;
320 	}

322 	/**
323 	 * Appends the specified visual properties to the current matrix.
324 	 * @param alpha desired alpha value
325 	 * @param shadow desired shadow value
326 	 * @param compositeOperation desired composite operation value
327 	 */
328 	Matrix2D.prototype.appendProperties = function(alpha, shadow, compositeOperation) {
329 		this.alpha *= alpha;
330 		this.shadow = shadow || this.shadow;
331 		this.compositeOperation = compositeOperation || this.compositeOperation;
332 	}

334 	/**
335 	 * Prepends the specified visual properties to the current matrix.
336 	 * @param alpha desired alpha value
337 	 * @param shadow desired shadow value
338 	 * @param compositeOperation desired composite operation value
339 	 */
340 	Matrix2D.prototype.prependProperties = function(alpha, shadow, compositeOperation) {
341 		this.alpha *= alpha;
342 		this.shadow = this.shadow || shadow;
343 		this.compositeOperation = this.compositeOperation || compositeOperation;
344 	}

346 	/**
347 	* Returns a clone of this Matrix.
348 	**/
349 	Matrix2D.prototype.clone = function() {
350 		var mtx = new Matrix2D(this.a, this.b, this.c, this.d, this.tx, this.ty);
351 		mtx.shadow = this.shadow;
352 		mtx.alpha = this.alpha;
353 		mtx.compositeOperation = this.compositeOperation;
354 		return mtx;
355 	}
