A sprite sheet is a series of images (usually animation frames) combined into a single image on a regular grid. For example, an animation consisting of 8 100x100 images could be combined into a 400x200 sprite sheet (4 frames across by 2 high).

 41 	/** @private */
 42 	SpriteSheetUtils._workingCanvas = document.createElement("canvas");
 43 	/** @private */
 44 	SpriteSheetUtils._workingContext = SpriteSheetUtils._workingCanvas.getContext("2d");

 46 // public static methods:
 47 	/**
 48 	 * Builds a new extended sprite sheet based on the specified sprite sheet by adding flipped frames
 49 	 * (vertical, horizontal, or both). Flipping elements on the display list by using setting scaleX/scaleY
 50 	 * to -1 is quite expensive in most browsers, so this method allows you to incur the cost of flipping once,
 51 	 * in advance, without increasing the load size of your sprite sheets. Returns a new SpriteSheet instance
 52 	 * containing the generated image and frame data.
 53 	 * @param spriteSheet The sprite sheet to use as the source.
 54 	 * @param flipData A generic object that specifies which frames will be flipped, what to name the flipped result, and how to flip the frames (horizontally, vertically, or both). Each property name indicates the name of a new sequence to create, and should reference an array where the first index is the name of the original sequence to flip, the second index indicates whether to flip it horizontally, the third index indicates whether to flip it vertically, and the fourth indicates what the "next" value for the resulting frame data should be. For example, the following would create a new sequence named "walk_left" consisting of the frames from the original "walk_right" sequence flipped horizontally: {walk_left:["walk_right", true, false]}
 55 	 * @static
 56 	 **/
 57 	SpriteSheetUtils.flip = function(spriteSheet, flipData) {
 58 		var image = spriteSheet.image;
 59 		var frameData = spriteSheet.frameData;
 60 		var frameWidth = spriteSheet.frameWidth;
 61 		var frameHeight = spriteSheet.frameHeight;
 62 		
 63 		var cols = image.width/frameWidth|0;
 64 		var rows = image.height/frameHeight|0;
 65 		var ttlFrames = cols*rows;
 66 		
 67 		// clone frameData:
 68 		var frData = {};
 69 		var data;
 70 		for (var n in frameData) {
 71 			data = frameData[n];
 72 			if (data instanceof Array) { data = data.slice(0); }
 73 			frData[n] = data;
 74 		}
 75 		
 76 		// calculate how many new frames we're generating, and build a map:
 77 		var map = [];
 78 		var frCount = 0;
 79 		var i = 0;
 80 		for (n in flipData) {
 81 			var fd = flipData[n];
 82 			data = frameData[fd[0]];
 83 			if (data ==  null) { continue; }
 84 			if (data instanceof Array) {
 85 				var start = data[0];
 86 				var end = data[1];
 87 				if (end == null) { end = start; }
 88 			} else {
 89 				start = end = data;
 90 			}
 91 			map[i] = n;
 92 			map[i+1] = start;
 93 			map[i+2] = end;
 94 			frCount += end-start+1;
 95 			i+=4;
 96 		}
 97 		
 98 		// get the canvas ready:
 99 		var canvas = SpriteSheetUtils._workingCanvas;
100 		canvas.width = image.width;
101 		canvas.height = Math.ceil(rows+frCount/cols)*frameHeight;
102 		var ctx = SpriteSheetUtils._workingContext;
103 		ctx.drawImage(image, 0, 0, cols*frameWidth, rows*frameHeight, 0, 0, cols*frameWidth, rows*frameHeight);
104 		
105 		// draw the new frames, and update the new frameData:
106 		var frame = ttlFrames-1;
107 		for (i=0; i<map.length; i+=4) {
108 			n = map[i];
109 			start = map[i+1];
110 			end = map[i+2];
111 			fd = flipData[n];
112 			var flipH = fd[1]?-1:1;
113 			var flipV = fd[2]?-1:1;
114 			var offH = flipH==-1?frameWidth:0;
115 			var offV = flipV==-1?frameHeight:0;
116 			
117 			for (var j=start; j<=end; j++) {
118 				frame++;
119 				ctx.save();
120 				ctx.translate((frame%cols)*frameWidth+offH, (frame/cols|0)*frameHeight+offV);
121 				ctx.scale(flipH, flipV);
122 				ctx.drawImage(image, (j%cols)*frameWidth, (j/cols|0)*frameHeight, frameWidth, frameHeight, 0, 0, frameWidth, frameHeight);
123 				
124 				ctx.restore();
125 			}
126 			frData[n] = [frame-(end-start), frame, fd[3]];
127 			
128 		}
129 		
130 		var img = new Image();
131 		img.src = canvas.toDataURL("image/png");
132 		return new SpriteSheet((img.width > 0)? img : canvas, frameWidth, frameHeight, frData);
133 	}

135 	/**
136 	* Returns a string representing the specified frameData object.
137 	* @param frameData The frame data to output.
138 	* @static
139 	**/
140 	SpriteSheetUtils.frameDataToString = function(frameData) {
141 		var str = "";
142 		var max = 0;
143 		var min = 0;
144 		var count = 0;
145 		var data, next;
146 		for (var n in frameData) {
147 			count++;
148 			data = frameData[n];
149 			if (data instanceof Array) {
150 				var start = data[0];
151 				var end = data[1];
152 				if (end == null) { end = start; }
153 				next = data[2];
154 				if (next == null) { next = n; }
155 			} else {
156 				start = end = data;
157 				next = n;
158 			}
159 			str += "\n\t"+n+", start="+start+", end="+end+", next="+next;
160 			if (next == false) { str += " (stop)"; }
161 			else if (next == n) { str += " (loop)"; }
162 			if (end > max) { max = end; }
163 			if (start < min) { min = start; }
164 		}
165 		str = count+" sequences, min="+min+", max="+max+str;
166 		return str;
167 	}

169 	/**
170 	 * Returns a single frame of the specified sprite sheet as a new PNG image.
171 	 * @param spriteSheet The SpriteSheet instance to extract a frame from.
172 	 * @param frame The frame number or sequence name to extract. If a sequence name is specified, only the first frame of the sequence will be extracted.
173 	 */
174 	SpriteSheetUtils.extractFrame = function(spriteSheet, frame) {
175 		var image = spriteSheet.image;
176 		var frameWidth = spriteSheet.frameWidth;
177 		var frameHeight = spriteSheet.frameHeight;
178 		var cols = image.width/frameWidth|0;
179 		if (isNaN(frame)) {
180 			var data = spriteSheet.frameData[frame];
181 			if (data instanceof Array) { frame = data[0]; }
182 			else { frame = data; }
183 		}
184 		var canvas = SpriteSheetUtils._workingCanvas;
185 		canvas.width = frameWidth;
186 		canvas.height = frameHeight;
187 		SpriteSheetUtils._workingContext.drawImage(image, (frame%cols)*frameWidth, (frame/cols|0)*frameHeight, frameWidth, frameHeight, 0, 0, frameWidth, frameHeight);
188 		var img = new Image();
189 		img.src = canvas.toDataURL("image/png");
190 		return img;
191 	}
