 32 /**
 33 * Constructs a BitmapSequence object with the specified sprite sheet.
 34 * @param spriteSheet The SpriteSheet instance to play back. This includes the source image, frame dimensions, and frame data. See SpriteSheet for more information.
 35 * @class Displays frames or sequences of frames from a sprite sheet image. A sprite sheet is a series of images (usually animation frames) combined into a single image on a regular grid. For example, an animation consisting of 8 100x100 images could be combined into a 400x200 sprite sheet (4 frames across by 2 high). You can display individual frames, play sequential frames as an animation, and even sequence animations together. See the SpriteSheet class for more information on setting up frames and animation.
 36 * @augments DisplayObject
 37 */
 38 function BitmapSequence(spriteSheet) {
 39   this.initialize(spriteSheet);
 40 this.prototype = new DisplayObject();
 41 
 42 // public properties:
 43 	/** Specifies a funciton to call whenever any sequence reaches its end. */
 44 	this.callback = null;
 45 	/** The frame that will be drawn on the next tick. This can also be set, but it will not update the current sequence, so it may result in unexpected behaviour if you are using frameData. */
 46 	this.currentFrame = -1;
 47 	/** Returns the currently playing sequence when using frameData. READ-ONLY. */
 48 	this.currentSequence = null; // READ-ONLY
 49 	/** Returns the last frame of the currently playing sequence when using frameData. READ-ONLY. */
 50 	this.currentEndFrame = null; // READ-ONLY
 51 	/** Returns the first frame of the currently playing sequence when using frameData. READ-ONLY. */
 52 	this.currentStartFrame = null; // READ-ONLY
 53 	/** Returns the name of the next sequence that will be played, or null if it will stop playing after the current sequence. READ-ONLY. */
 54 	this.nextSequence = null;
 55 	/** Prevents the animation from advancing each tick automatically. For example, you could create a sprite sheet of icons, set paused to true, and display the appropriate icon by setting currentFrame. */
 56 	this.paused = false;
 57 	/** The SpriteSheet instance to play back. This includes the source image, frame dimensions, and frame data. See SpriteSheet for more information. */
 58 	this.spriteSheet = null;
 59 	this.snapToPixel = true;
 60 	
 61 // constructor:
 62 	/** @private */
 63 	this.DisplayObject_initialize = this.initialize;
 64 	/** @private */
 65 	this.initialize = function(spriteSheet) {
 66 		this.DisplayObject_initialize();
 67 		this.spriteSheet = spriteSheet;
 68 	}
 69 	
 70 // public methods:
 71 	this.isVisible = function() {
 72 		var image = this.spriteSheet ? this.spriteSheet.image : null;
 73 		return this.visible && this.alpha > 0 && this.scaleX != 0 && this.scaleY != 0 && image && this.currentFrame >= 0 && (image.complete || image.getContext);
 74 	}
 75 
 76 	this.DisplayObject_draw = this.draw;
 77 	this.draw = function(ctx, ignoreCache) {
 78 		if (this.DisplayObject_draw(ctx,ignoreCache)) { return true; }
 79 		
 80 		var image = this.spriteSheet.image;
 81 		var frameWidth = this.spriteSheet.frameWidth;
 82 		var frameHeight = this.spriteSheet.frameHeight;
 83 		var cols = image.width/frameWidth|0;
 84 		var rows = image.height/frameHeight|0;
 85 		
 86 		if (this.currentEndFrame != null) {
 87 			if (this.currentFrame > this.currentEndFrame) {
 88 				if (this.nextSequence) {
 89 					this._goto(this.nextSequence);
 90 				} else {
 91 					this.paused = true;
 92 					this.currentFrame = this.currentEndFrame;
 93 				}
 94 				if (this.callback) { this.callback(this); }
 95 			}
 96 		} else if (this.spriteSheet.frameData) {
 97 			// sequence data is set, but we haven't actually played a sequence yet:
 98 			this.paused = true;
 99 		} else {
100 			var ttlFrames = this.spriteSheet.totalFrames || cols*rows;
101 			if (this.currentFrame >= ttlFrames) {
102 				if (this.spriteSheet.loop) { this.currentFrame = 0; }
103 				else {
104 					this.currentFrame = ttlFrames;
105 					this.paused = true;
106 				}
107 				if (this.callback) { this.callback(this); }
108 			}
109 		}
110 		if (this.currentFrame >= 0) {
111 			var col = this.currentFrame%cols;
112 			var row = this.currentFrame/cols|0;
113 			ctx.drawImage(image, frameWidth*col, frameHeight*row, frameWidth, frameHeight, 0, 0, frameWidth, frameHeight);
114 		}
115 		return true;
116 	}
117 	
118 	/**
119 	* Advances the currentFrame if paused is not true. This is called automatically when the Stage ticks.
120 	*/
121 	this.tick = function() {
122 		if (this.paused) { return; }
123 		this.currentFrame++;
124 	}
125 	
126 	/**
127 	* Because the content of a Bitmap is already in a simple format, cache is unnecessary for BitmapSequence instances.
128 	*/
129 	this.cache = function() {}
130 	/**
131 	* Because the content of a Bitmap is already in a simple format, cache is unnecessary for BitmapSequence instances.
132 	*/
133 	this.uncache = function() {}
134 	
135 	/**
136 	* Sets paused to false and plays the specified sequence name, named frame, or frame number.
137 	*/
138 	this.gotoAndPlay = function(frameOrSequence) {
139 		this.paused = false;
140 		this._goto(frameOrSequence);
141 	}
142 	
143 	/**
144 	* Sets paused to true and seeks to the specified sequence name, named frame, or frame number.
145 	*/
146 	this.gotoAndStop = function(frameOrSequence) {
147 		this.paused = true;
148 		this._goto(frameOrSequence);
149 	}
150 	
151 	this.clone = function() {
152 		var o = new BitmapSequence(this.spriteSheet);
153 		this.cloneProps(o);
154 		return o;
155 	}
156 		
157 	this.toString = function() {
158 		return "[BitmapSequence (name="+  this.name +")]";
159 	}
160 	
161 // private methods:
162 	/** @private */
163 	this.DisplayObject_cloneProps = this.cloneProps;
164 	/** @private */
165 	this.cloneProps = function(o) {
166 		this.DisplayObject_cloneProps(o);
167 		o.callback = this.callback;
168 		o.currentFrame = this.currentFrame;
169 		o.currentStartFrame = this.currentStartFrame;
170 		o.currentEndFrame = this.currentEndFrame;
171 		o.currentSequence = this.currentSequence;
172 		o.nextSequence = this.nextSequence;
173 		o.paused = this.paused;
174 		o.frameData = this.frameData;
175 	}
176 	
177 	/** @private */
178 	this._goto = function(frameOrSequence) {
179 		if (isNaN(frameOrSequence)) {
180 			if (frameOrSequence == this.currentSequence) {
181 				this.currentFrame = this.currentStartFrame;
182 				return;
183 			}
184 			var data = this.spriteSheet.frameData[frameOrSequence];
185 			if (data instanceof Array) {
186 				this.currentFrame = this.currentStartFrame = data[0];
187 				this.currentSequence = frameOrSequence;
188 				this.currentEndFrame = data[1];
189 				if (this.currentEndFrame == null) { this.currentEndFrame = this.currentStartFrame; }
190 				if (this.currentEndFrame == null) { this.currentEndFrame = this.currentFrame; }
191 				this.nextSequence = data[2];
192 				if (this.nextSequence == null) { this.nextSequence = this.currentSequence; }
193 				else if (this.nextSequence == false) { this.nextSequence = null; }
194 			} else {
195 				this.currentSequence = this.nextSequence = null;
196 				this.currentEndFrame = this.currentFrame = this.currentStartFrame = data;
197 			}
198 		} else {
199 			this.currentSequence = this.nextSequence = this.currentEndFrame = null;
200 			this.currentStartFrame = 0;
201 			this.currentFrame = frameOrSequence;
202 		}
203 	}
204 }
