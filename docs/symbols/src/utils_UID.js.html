 32 // constructor:
 33 	/**
 34 	* The UID class uses a static interface (ex. UID.get()) and should not be instantiated.
 35 	* @class Global utility for generating sequential unique ID numbers.
 36 	*/
 37 	function UID() {

 41 // private static properties:
 42 	/** @private */
 43 	UID._nextID = 0;
 44 	
 45 	/**
 46 	* Returns the next unique id.
 47 	* @static
 48 	*/
 49 	UID.get = function() {
 50 		return UID._nextID++;
 51 	}
